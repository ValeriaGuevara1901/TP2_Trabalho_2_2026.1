29char* BCD_ConverterLongASCII(int dimASCII,
30 char* pNumASCII,
31 long Numero) {
32 // Assertivas de entrada
33 assert(pNumASCII != NULL);
34 assert(dimASCII >= 2);
35
36 if (pNumASCII == NULL || dimASCII < 2) {
37 return NULL;
38 }
39
40 // Converte para string usando snprintf (seguro)
41 char buffer[64];
42 snprintf(buffer, sizeof(buffer), "%ld", Numero);
43
44 // Copia com truncagem à direita se necessário
45 int tamanho = static_cast<int>(strlen(buffer));
46 int copiar = (tamanho < dimASCII - 1) ? tamanho : dimASCII - 1;
47 strncpy(pNumASCII, buffer, copiar);
48 pNumASCII[copiar] = '\0';
49
50 // Assertiva de saída: primeiro char é '-' sse Numero < 0
51 assert((Numero < 0) ? (pNumASCII[0] == '-') : (pNumASCII[0] != '-'));
52 // Assertiva de saída: string terminada em '\0'
53 assert(pNumASCII[copiar] == '\0');
54
55 return pNumASCII;
56}

71bool LOG_ParseLinha(const std::string& linha, RegistroLog* registro) {
72 // Assertivas de entrada
73 assert(registro != NULL);
74 assert(!linha.empty());
75
76 if (registro == NULL || linha.empty()) {
77 return false;
78 }
79
80 // Formato esperado: "DD/MM/AAAA HH:MM:SS mensagem"
81 // Expressão regular equivalente: ^(\d{1,2})\/(\d{1,2})\/(\d{4}) (\d{2}):(\d{2}):(\d{2}) (.+)$
82 int dia, mes, ano, hora, minuto, segundo;
83 char mensagem[MAX_MSG_LOG];
84 memset(mensagem, 0, sizeof(mensagem));
85
86 int lidos = sscanf(linha.c_str(), // NOLINT
87 "%d/%d/%d %d:%d:%d %100[^\n]",
88 &dia, &mes, &ano,
89 &hora, &minuto, &segundo,
90 mensagem);
91
92 if (lidos < 6) {
93 return false;
94 }
95
96 // Validação dos campos de data/hora
97 if (dia < 1 || dia > 31) return false;
98 if (mes < 1 || mes > 12) return false;
99 if (ano < 1900 || ano > 9999) return false;
100 if (hora < 0 || hora > 23) return false;
101 if (minuto < 0 || minuto > 59) return false;
102 if (segundo < 0 || segundo > 59) return false;
103
104 registro->dia = dia;
105 registro->mes = mes;
106 registro->ano = ano;
107 registro->hora = hora;
108 registro->minuto = minuto;
109 registro->segundo = segundo;
110 strncpy(registro->mensagem, mensagem, MAX_MSG_LOG - 1);
111 registro->mensagem[MAX_MSG_LOG - 1] = '\0';
112
113 // Assertivas de saída
114 assert(registro->dia >= 1 && registro->dia <= 31);
115 assert(registro->mes >= 1 && registro->mes <= 12);
116 assert(registro->hora >= 0 && registro->hora <= 23);
117 assert(registro->minuto >= 0 && registro->minuto <= 59);
118 assert(registro->segundo >= 0 && registro->segundo <= 59);
119
120 return true;
121}

127std::string LOG_ObterNomeArquivoTotal(const std::string& caminhoLog) {
128 // Assertiva de entrada
129 assert(!caminhoLog.empty());
130
131 // Encontra a última barra (Windows ou Unix)
132 size_t posBarraWin = caminhoLog.rfind('\\');
133 size_t posBarraUnix = caminhoLog.rfind('/');
134
135 size_t posBarra = std::string::npos;
136 if (posBarraWin != std::string::npos && posBarraUnix != std::string::npos) {
137 posBarra = std::max(posBarraWin, posBarraUnix);
138 } else if (posBarraWin != std::string::npos) {
139 posBarra = posBarraWin;
140 } else if (posBarraUnix != std::string::npos) {
141 posBarra = posBarraUnix;
142 }
143
144 // Prefixa "total_" ao nome do arquivo
145 if (posBarra == std::string::npos) {
146 // Sem diretório, só o nome do arquivo
147 return "total_" + caminhoLog;
148 } else {
149 // Com diretório
150 std::string diretorio = caminhoLog.substr(0, posBarra + 1);
151 std::string nomeArquivo = caminhoLog.substr(posBarra + 1);
152 return diretorio + "total_" + nomeArquivo;
153 }
154}

160int LOG_CompararRegistros(const RegistroLog& a, const RegistroLog& b) {
161 // Compara ano
162 if (a.ano != b.ano) return a.ano - b.ano;
163 // Compara mês
164 if (a.mes != b.mes) return a.mes - b.mes;
165 // Compara dia
166 if (a.dia != b.dia) return a.dia - b.dia;
167 // Compara hora
168 if (a.hora != b.hora) return a.hora - b.hora;
169 // Compara minuto
170 if (a.minuto != b.minuto) return a.minuto - b.minuto;
171 // Compara segundo
172 return a.segundo - b.segundo;
173}

179std::vector<RegistroLog> LOG_MergeOrdenado(
180 const std::vector<RegistroLog>& base,
181 const std::vector<RegistroLog>& novos) {
182
183 // Combina os dois vetores
184 std::vector<RegistroLog> resultado;
185 resultado.reserve(base.size() + novos.size());
186 resultado.insert(resultado.end(), base.begin(), base.end());
187 resultado.insert(resultado.end(), novos.begin(), novos.end());
188
189 // Ordena por data/hora usando LOG_CompararRegistros
190 std::sort(resultado.begin(), resultado.end(),
191 [](const RegistroLog& a, const RegistroLog& b) {
192 return LOG_CompararRegistros(a, b) < 0;
193 });
194
195 return resultado;
196}

202std::vector<RegistroLog> LOG_LerArquivoLog(const std::string& caminhoLog) {
203 std::vector<RegistroLog> registros;
204
205 std::ifstream arquivo(caminhoLog);
206 if (!arquivo.is_open()) {
207 return registros; // Arquivo não existe ou não pode ser aberto
208 }
209
210 std::string linha;
211 while (std::getline(arquivo, linha)) {
212 RegistroLog reg;
213 if (LOG_ParseLinha(linha, &reg)) {
214 registros.push_back(reg);
215 }
216 // Linhas inválidas são ignoradas
217 }
218
219 arquivo.close();
220 return registros;
221}

228 const std::vector<RegistroLog>& registros) {
229 std::ofstream arquivo(caminhoLog);
230 if (!arquivo.is_open()) {
231 return false;
232 }
233
234 for (const auto& reg : registros) {
235 arquivo << reg.dia << "/" << reg.mes << "/" << reg.ano << " "
236 << std::setfill('0') << std::setw(2) << reg.hora << ":"
237 << std::setfill('0') << std::setw(2) << reg.minuto << ":"
238 << std::setfill('0') << std::setw(2) << reg.segundo << " "
239 << reg.mensagem << "\n";
240 }
241
242 arquivo.close();
243 return true;
244}

250bool LOG_ProcessarArquivo(const std::string& caminhoLog) {
251 // Lê os registros do arquivo de log
252 auto novosRegistros = LOG_LerArquivoLog(caminhoLog);
253 if (novosRegistros.empty()) {
254 // Arquivo não existe ou está vazio
255 std::ifstream teste(caminhoLog);
256 if (!teste.is_open()) {
257 return false; // Arquivo não existe
258 }
259 teste.close();
260 }
261
262 // Obtém o caminho do arquivo total
263 std::string caminhoTotal = LOG_ObterNomeArquivoTotal(caminhoLog);
264
265 // Lê os registros existentes do arquivo total (se existir)
266 auto registrosExistentes = LOG_LerArquivoLog(caminhoTotal);
267
268 // Faz merge dos registros
269 auto resultado = LOG_MergeOrdenado(registrosExistentes, novosRegistros);
270
271 // Escreve o resultado no arquivo total
272 return LOG_EscreverArquivoLog(caminhoTotal, resultado);
273}

280 // Lê a lista de arquivos a monitorar

287 return -1; // Arquivo de configuração não existe
288 }
289 teste.close();
290 return 0; // Arquivo existe mas está vazio
291 }
292
293 // Processa cada arquivo
294 int processados = 0;
295 for (const auto& arquivo : arquivos) {
296 if (LOG_ProcessarArquivo(arquivo)) {
297 processados++;
298 }

300
